 15 /**
 16  * @class
 17  * GeoRSS loader: Load GeoRSS feeds.
 18  *
 19  * <p> Parsing is complicated by the 
 20  * diversity of GeoRSS formats; this parser handles:</p>
 21  * <ul>
 22  *  <li>RSS feeds</li>
 23  *  <li>Atom feeds</li>
 24  * </ul>
 25  * <p>and looks for geographic information in the following formats:</p>
 26  * <ul>
 27  *  <li>GeoRSS-Simple</li>
 28  *  <li>GML </li>
 29  *  <li>W3C Geo</li>
 30  * </ul>
 31  * <p>At the moment, this only supports points, polygons, polylines; boxes
 32  * may be added at some later point.</p>
 33  *
 34  * @augments TimeMap.loaders.xml
 35  * @requires loaders/xml.js
 36  * @requires param.js
 37  * @borrows TimeMap.loaders.georss.parse as #parse
 38  *
 39  * @example
 40 TimeMap.init({
 41     datasets: [
 42         {
 43             title: "GeoRSS Dataset",
 44             type: "georss", // Data to be loaded in GeoRSS
 45             options: {
 46                 url: "mydata.rss" // GeoRSS file to load - must be a local URL
 47             }
 48         }
 49     ],
 50     // etc...
 51 });
 52  * @see <a href="../../examples/earthquake_georss.html">GeoRSS Example</a>
 53  *
 54  * @param {Object} options          All options for the loader:
 55  * @param {String} options.url          URL of GeoRSS file to load (NB: must be local address)
 56  * @param {mixed} [options[...]]        Other options (see {@link TimeMap.loaders.xml})
 57  */
 58 TimeMap.loaders.georss = function(options) {
 59     var loader = new TimeMap.loaders.xml(options);
 60     loader.parse = TimeMap.loaders.georss.parse;
 61     return loader;
 62 };
 63 
 64 /**
 65  * Static function to parse GeoRSS
 66  *
 67  * @param {XML} node      GeoRSS node to be parsed
 68  * @return {TimeMapItem[]}  Array of TimeMapItems
 69  */
 70 TimeMap.loaders.georss.parse = function(node) {
 71     var items = [], data, placemarks, pm, i, nList;
 72     
 73     // get TimeMap utilty functions
 74     // assigning to variables should compress better
 75     var util = TimeMap.util,
 76         getTagValue = util.getTagValue,
 77         getNodeList = util.getNodeList,
 78         makePoint = util.makePoint,
 79         makePoly = util.makePoly,
 80         formatDate = util.formatDate,
 81         nsMap = util.nsMap;
 82     
 83     // define namespaces
 84     nsMap.georss = 'http://www.georss.org/georss';
 85     nsMap.gml = 'http://www.opengis.net/gml';
 86     nsMap.geo = 'http://www.w3.org/2003/01/geo/wgs84_pos#';
 87     nsMap.kml = 'http://www.opengis.net/kml/2.2';
 88     
 89     // determine whether this is an Atom feed or an RSS feed
 90     var feedType = (node.firstChild.tagName == 'rss') ? 'rss' : 'atom';
 91     
 92     // look for placemarks
 93     var tName = (feedType == 'rss' ? "item" : "entry");
 94     placemarks = getNodeList(node, tName);
 95     for (i=0; i<placemarks.length; i++) {
 96         pm = placemarks[i];
 97         data = { options: {} };
 98         // get title & description
 99         data.title = getTagValue(pm, "title");
100         tName = (feedType == 'rss' ? "description" : "summary");
101         data.options.description = getTagValue(pm, tName);
102         // get time information, allowing KML-namespaced time elements
103         nList = getNodeList(pm, "TimeStamp", "kml");
104         if (nList.length > 0) {
105             data.start = getTagValue(nList[0], "when", "kml");
106         }
107         // look for timespan
108         if (!data.start) {
109             nList = getNodeList(pm, "TimeSpan", "kml");
110             if (nList.length > 0) {
111                 data.start = getTagValue(nList[0], "begin", "kml");
112                 data.end = getTagValue(nList[0], "end", "kml") ||
113                     // unbounded spans end at the present time
114                     formatDate(new Date());
115             }
116         }
117         // otherwise, use pubDate/updated elements
118         if (!data.start) {
119             if (feedType == 'rss') {
120                 // RSS needs date conversion
121                 var d = new Date(Date.parse(getTagValue(pm, "pubDate")));
122                 // reformat
123                 data.start = formatDate(d);
124             } else {
125                 // atom uses ISO 8601
126                 data.start = getTagValue(pm, "updated");
127             }
128         }
129         // find placemark - single geometry only for the moment
130         var done = false;
131         PLACEMARK: while (!done) {
132             var coords, geom;
133             // look for point, GeoRSS-Simple
134             coords = getTagValue(pm, "point", 'georss');
135             if (coords) {
136                 data.point = makePoint(coords); 
137                 break PLACEMARK;
138             }
139             // look for point, GML
140             nList = getNodeList(pm, "Point", 'gml');
141             if (nList.length > 0) {
142                 // GML <pos>
143                 coords = getTagValue(nList[0], "pos", 'gml');
144                 // GML <coordinates>
145                 if (!coords) {
146                     coords = getTagValue(nList[0], "coordinates", 'gml');
147                 }
148                 if (coords) {
149                     data.point = makePoint(coords); 
150                     break PLACEMARK;
151                 }
152             }
153             // look for point, W3C Geo
154             if (getTagValue(pm, "lat", 'geo')) {
155                 coords = [
156                     getTagValue(pm, "lat", 'geo'),
157                     getTagValue(pm, "long", 'geo')
158                 ];
159                 data.point = makePoint(coords); 
160                 break PLACEMARK;
161             }
162             // look for polyline, GeoRSS-Simple
163             coords = getTagValue(pm, "line", 'georss');
164             if (coords) {
165                 data.polyline = makePoly(coords); 
166                 break PLACEMARK;
167             }
168             // look for polygon, GeoRSS-Simple
169             coords = getTagValue(pm, "polygon", 'georss');
170             if (coords) {
171                 data.polygon = makePoly(coords); 
172                 break PLACEMARK;
173             }
174             // look for polyline, GML
175             nList = getNodeList(pm, "LineString", 'gml');
176             if (nList.length > 0) {
177                 geom = "polyline";
178             } else {
179                 nList = getNodeList(pm, "Polygon", 'gml');
180                 if (nList.length > 0) {
181                     geom = "polygon";
182                 }
183             }
184             if (nList.length > 0) {
185                 // GML <posList>
186                 coords = getTagValue(nList[0], "posList", 'gml');
187                 // GML <coordinates>
188                 if (!coords) {
189                     coords = getTagValue(nList[0], "coordinates", 'gml');
190                 }
191                 if (coords) {
192                     data[geom] = makePoly(coords);
193                     break PLACEMARK;
194                 }
195             }
196             
197             // XXX: deal with boxes
198             
199             done = true;
200         }
201         // look for any extra tags specified
202         this.parseExtra(data, pm);
203         items.push(data);
204     }
205     
206     // clean up
207     node = placemarks = pm = nList = null;
208     return items;
209 };
